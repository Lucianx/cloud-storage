        $("dd a").click(function () {
            var search = $(this).attr("href").split("?")[1];
            $("form input[type=hidden][id]").each(function () {
                var name = $(this).attr("name");
                var val = QueryString(name, search);
                if (!val) return;
                $(this).val(val);
            });
            $("form").submit();
            return false;
        });
